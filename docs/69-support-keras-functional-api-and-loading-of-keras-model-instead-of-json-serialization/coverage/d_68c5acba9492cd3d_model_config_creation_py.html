24def physXAI_2_agentlib_json(run_id: str, preprocessing_dict: dict, model_dict: dict = None, training_dict: dict = None, 

25 model_type: str = 'ANN') -> dict: 

26 """ 

27 Converts physXAI model configurations to an AgentLib-MPC compatible JSON format. 

28 Args: 

29 run_id (str): The unique identifier for the mpc run. 

30 preprocessing_dict (dict): The preprocessing configuration from physXAI. 

31 model_dict (dict, optional): The model configuration from physXAI. Defaults to None. 

32 training_dict (dict, optional): The training configuration from physXAI. Defaults to None. 

33 model_type (str, optional): The type of model ('ANN' or 'LinReg'). Defaults to 'ANN'. 

34 Returns: 

35 dict: The converted configuration in AgentLib-MPC JSON format. 

36 """ 

37 

38 # Initialize target dictionary 

39 target_dict = { 

40 "dt": preprocessing_dict["time_step"], 

41 "input": {}, 

42 "output": {}, 

43 "agentlib_mpc_hash": f"physXAI", 

44 "training_info": {"preprocessing": {}, "model": {}, "training": {}}, 

45 } 

46 

47 # Preprocessing Info 

48 for info in preprocessing_training_info: 

49 if info in preprocessing_dict: 

50 target_dict["training_info"]["preprocessing"][info] = preprocessing_dict[info] 

51 

52 # Model Info 

53 if model_dict is not None: 

54 target_dict["training_info"]["model"] = model_dict 

55 

56 # Training Info 

57 if training_dict is not None: 

58 target_dict["training_info"]["training"] = training_dict 

59 

60 # Validate and extract shift and output 

61 default_shift = preprocessing_dict.get("shift", 1) 

62 if default_shift != 1: 

63 raise ValueError(f"Config Translation Error: Shift should be 1 to be used in AgentLib, bus was {default_shift}") 

64 if not isinstance(preprocessing_dict.get("output"), list) or len(preprocessing_dict["output"]) != 1: 

65 raise ValueError("Config Translation Error: Output should be a list with 1 element") 

66 

67 # Extract output type 

68 output_str = preprocessing_dict["output"][0] 

69 output_type = "absolute" 

70 output_full_feature_name = output_str 

71 change_match = re.match(output_type_pattern, output_str) 

72 if change_match: 

73 output_type = "difference" 

74 output_full_feature_name = change_match.group(1).strip() 

75 output_key_name = output_full_feature_name 

76 

77 # Extract inputs 

78 grouped_inputs = defaultdict(list) 

79 for i, input_str in enumerate(preprocessing_dict["inputs"]): 

80 lag = default_shift 

81 feature_base_str = input_str 

82 

83 lag_match = re.search(lag_pattern, input_str) 

84 if lag_match: 

85 lag_value = int(lag_match.group(1)) 

86 lag = default_shift + lag_value 

87 feature_base_str = input_str[:lag_match.start()] 

88 base_name = feature_base_str 

89 

90 grouped_inputs[base_name].append({'original_index': i, 'lag': lag, 'full_name': input_str}) 

91 

92 # Validate and extract lags 

93 for base_name, items in grouped_inputs.items(): 

94 if len(items) > 1: 

95 items.sort(key=lambda x: x['original_index']) 

96 for j in range(len(items) - 1): 

97 current_item = items[j] 

98 next_item = items[j + 1] 

99 if next_item['original_index'] != current_item['original_index'] + 1: 

100 raise ValueError( 

101 f"Config Translation Error: Features for '{base_name}' are not grouped consecutively. " 

102 f"Found '{current_item['full_name']}' at Index {current_item['original_index']} and " 

103 f"'{next_item['full_name']}' at Index {next_item['original_index']}." 

104 ) 

105 

106 if next_item['lag'] != current_item['lag'] + 1: 

107 raise ValueError( 

108 f"Config Translation Error: Lags for '{base_name}' are not in ascending order. " 

109 f"Lag {current_item['lag'] - 1} is followed by {next_item['lag'] - 1}." 

110 ) 

111 

112 # Generate final inputs 

113 for base_name, items in grouped_inputs.items(): 

114 max_lag = max(item['lag'] for item in items) 

115 target_dict["input"][base_name] = { 

116 "name": base_name, 

117 "lag": max_lag 

118 } 

119 

120 # Validate and extract output 

121 recursive = False 

122 num_recursive_inputs = 1 

123 if output_key_name in target_dict["input"]: 

124 recursive = True 

125 

126 recursive_inputs = grouped_inputs[output_key_name] 

127 num_recursive_inputs = len(recursive_inputs) 

128 total_inputs = len(preprocessing_dict["inputs"]) 

129 expected_indices = list(range(total_inputs - num_recursive_inputs, total_inputs)) 

130 actual_indices = [item['original_index'] for item in recursive_inputs] 

131 if expected_indices != actual_indices: 

132 raise ValueError( 

133 f"Config Translation Error: Recursive Feature '{output_key_name}' und its Lags must be at the end of the " 

134 f"'inputs'-List. Expected Indices: {expected_indices}, " 

135 f"Actual Indices: {actual_indices}." 

136 ) 

137 target_dict["input"].pop(output_key_name) 

138 

139 # Generate final output 

140 target_dict["output"][output_key_name] = { 

141 "name": output_key_name, 

142 "lag": num_recursive_inputs, 

143 "output_type": output_type, 

144 "recursive": recursive 

145 } 

146 

147 # LinReg is saved as normal AgentLib json with parameters, ANN as keras model path 

148 if model_type == 'LinReg' or (model_dict is not None and model_dict['__class_name__'] == 'LinearRegressionModel'): 

149 target_dict["model_type"] = "LinReg" 

150 

151 load_path = model_path_generation(run_id, output_key_name) + '.joblib' 

152 model = joblib.load(load_path) 

153 target_dict["parameters"] = { 

154 "coef": model.coef_.tolist(), 

155 "intercept": model.intercept_.tolist(), 

156 "n_features_in": model.n_features_in_, 

157 "rank": model.rank_, 

158 "singular": model.singular_.tolist(), 

159 } 

160 

161 else: 

162 target_dict["model_type"] = "KerasANN" 

163 target_dict["model_path"] = model_path_generation(run_id, output_key_name) + '.keras' 

164 

165 return target_dict
