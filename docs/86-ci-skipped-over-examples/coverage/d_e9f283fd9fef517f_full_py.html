42class DirectCollocation(basic.DirectCollocation): 

43 def _discretize(self, sys: FullSystem): 

44 """ 

45 Defines a direct collocation discretization. 

46 # pylint: disable=invalid-name 

47 """ 

48 

49 # setup the polynomial base 

50 collocation_matrices = self._collocation_polynomial() 

51 

52 # shorthands 

53 n = self.options.prediction_horizon 

54 ts = self.options.time_step 

55 

56 # Initial State 

57 x0 = self.add_opt_par(sys.initial_state) 

58 xk = self.add_opt_var(sys.states, lb=x0, ub=x0, guess=x0) 

59 uk = self.add_opt_par(sys.last_control) 

60 

61 # Parameters that are constant over the horizon 

62 const_par = self.add_opt_par(sys.model_parameters) 

63 du_weights = self.add_opt_par(sys.r_del_u) 

64 

65 # Formulate the NLP 

66 # loop over prediction horizon 

67 while self.k < n: 

68 # New NLP variable for the control 

69 u_prev = uk 

70 uk = self.add_opt_var(sys.controls) 

71 # penalty for control change between time steps 

72 self.objective_function += ts * ca.dot(du_weights, (u_prev - uk) ** 2) 

73 

74 # perform inner collocation loop 

75 opt_vars_inside_inner = [sys.algebraics, sys.outputs] 

76 opt_pars_inside_inner = [sys.non_controlled_inputs] 

77 

78 constant_over_inner = {sys.controls: uk, sys.model_parameters: const_par} 

79 xk_end, constraints = self._collocation_inner_loop( 

80 collocation=collocation_matrices, 

81 state_at_beginning=xk, 

82 states=sys.states, 

83 opt_vars=opt_vars_inside_inner, 

84 opt_pars=opt_pars_inside_inner, 

85 const=constant_over_inner, 

86 ) 

87 

88 # increment loop counter and time 

89 self.k += 1 

90 self.pred_time = ts * self.k 

91 

92 # New NLP variable for differential state at end of interval 

93 xk = self.add_opt_var(sys.states) 

94 

95 # Add continuity constraint 

96 self.add_constraint(xk - xk_end, gap_closing=True) 

97 

98 # add collocation constraints later for fatrop 

99 for constraint in constraints: 

100 self.add_constraint(*constraint)

103class MultipleShooting(basic.MultipleShooting): 

104 def _discretize(self, sys: FullSystem): 

105 """ 

106 Defines a multiple shooting discretization 

107 """ 

108 vars_dict = {sys.states.name: {}} 

109 n = self.options.prediction_horizon 

110 ts = self.options.time_step 

111 opts = {"t0": 0, "tf": ts} 

112 # Initial State 

113 x0 = self.add_opt_par(sys.initial_state) 

114 xk = self.add_opt_var(sys.states, lb=x0, ub=x0, guess=x0) 

115 vars_dict[sys.states.name][0] = xk 

116 uk = self.add_opt_par(sys.last_control) 

117 

118 # Parameters that are constant over the horizon 

119 du_weights = self.add_opt_par(sys.r_del_u) 

120 const_par = self.add_opt_par(sys.model_parameters) 

121 

122 # ODE is used here because the algebraics can be calculated with the stage function 

123 opt_integrator = self._create_ode(sys, opts, self.options.integrator) 

124 # initiate states 

125 while self.k < n: 

126 u_prev = uk 

127 uk = self.add_opt_var(sys.controls) 

128 # penalty for control change between time steps 

129 self.objective_function += ts * ca.dot(du_weights, (u_prev - uk) ** 2) 

130 dk = self.add_opt_par(sys.non_controlled_inputs) 

131 zk = self.add_opt_var(sys.algebraics) 

132 yk = self.add_opt_var(sys.outputs) 

133 

134 # get path constraints and objective values (stage) 

135 stage_arguments = { 

136 # variables 

137 sys.states.name: xk, 

138 sys.algebraics.name: zk, 

139 sys.outputs.name: yk, 

140 # parameters 

141 sys.controls.name: uk, 

142 sys.non_controlled_inputs.name: dk, 

143 sys.model_parameters.name: const_par, 

144 "__time": self.pred_time, 

145 } 

146 stage = self._stage_function(**stage_arguments) 

147 

148 # integral and multiple shooting constraint 

149 fk = opt_integrator( 

150 x0=xk, 

151 p=ca.vertcat(uk, dk, const_par, zk, yk), 

152 ) 

153 xk_end = fk["xf"] 

154 self.k += 1 

155 self.pred_time = ts * self.k 

156 xk = self.add_opt_var(sys.states) 

157 vars_dict[sys.states.name][self.k] = xk 

158 self.add_constraint(xk - xk_end, gap_closing=True) 

159 

160 # add model constraints last due to fatrop 

161 self.add_constraint( 

162 stage["model_constraints"], 

163 lb=stage["lb_model_constraints"], 

164 ub=stage["ub_model_constraints"], 

165 ) 

166 self.objective_function += stage["cost_function"] * ts
